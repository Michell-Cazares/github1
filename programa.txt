SubProceso TablaConRepetir ()
	Escribir "Ingrese un número"
	leer n
	Escribir "Valor inicial"
	Leer vi
	Escribir "Valor final"
	Leer vf
	a<-vi
	Repetir
		Escribir n " * " a " = " n*a
		a<-a+1
	Hasta Que a>vf
Fin SubProceso

SubProceso TablaConPara ()
	Escribir "Ingrese un número"
	leer n
	Escribir "Valor inicial"
	Leer vi
	Escribir "Valor final"
	Leer vf
	a<-vi
	Para a<-vi Hasta vf Con Paso 1 Hacer
		Escribir n " * " a " = " n*a
	Fin Para
Fin SubProceso

SubProceso TablaConMientras ()
	Escribir "Ingrese un número"
	leer n
	Escribir "Valor inicial"
	Leer vi
	Escribir "Valor final"
	Leer vf
	a<-vi
	Mientras a<=vf Hacer
		Escribir n " * " a " = " n*a
		a<-a+1
	FinMientras
Fin SubProceso
